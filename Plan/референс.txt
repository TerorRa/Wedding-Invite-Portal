function initTicket(){
  const splash=document.getElementById('splash');
  const wrap=document.getElementById('ticketWrap');
  const ticket=document.getElementById('ticketCard');
  if(!splash)return;
  document.body.style.overflow='hidden';
  document.documentElement.style.overflow='hidden';

  /* dot stars */
  for(let i=0;i<60;i++){
    const s=document.createElement('div');
    s.className='splash__dot';
    const sz=Math.random()*2.5+.8;
    const hi=(.5+Math.random()*.5).toFixed(2);
    s.style.cssText=`left:${Math.random()*100}%;top:${Math.random()*100}%;width:${sz}px;height:${sz}px;background:rgba(255,255,255,${hi});box-shadow:0 0 ${sz*2}px rgba(255,255,255,.4);--d:${2+Math.random()*4}s;--dl:${Math.random()*6}s;--lo:.1;--hi:${hi}`;
    splash.insertBefore(s,splash.firstChild);
  }
  /* 4-ray star shapes */
  for(let i=0;i<10;i++){
    const s=document.createElement('div');
    s.className='splash__star4';
    const sz=8+Math.random()*10;
    const hi=(.4+Math.random()*.4).toFixed(2);
    s.innerHTML=`<svg width="${sz}" height="${sz}" viewBox="0 0 20 20" fill="rgba(255,255,255,${hi})"><path d="M10 0L11.5 7.5L20 10L11.5 12.5L10 20L8.5 12.5L0 10L8.5 7.5Z"/></svg>`;
    s.style.cssText=`left:${Math.random()*100}%;top:${Math.random()*100}%;--d:${2+Math.random()*3}s;--dl:${Math.random()*5}s;--lo:.15;--hi:${hi}`;
    splash.insertBefore(s,splash.firstChild);
  }
  /* comets */
  for(let i=0;i<4;i++){
    const c=document.createElement('div');
    c.className='splash__comet';
    const w=50+Math.random()*100;
    c.style.cssText=`left:0;top:${3+Math.random()*50}%;width:${w}px;--cd:${3+Math.random()*4}s;--cdel:${i*1.5+Math.random()*2}s;--a:${20+Math.random()*15}deg;--dy:${10+Math.random()*20}vh`;
    splash.insertBefore(c,splash.firstChild);
  }


  /* hero starts hidden */
  const hero=document.getElementById('hero');
  if(hero)hero.classList.add('entering');

  /* dismiss */
  let opened=false;
  splash.addEventListener('click',()=>{
    if(opened)return;opened=true;
    /* ticket flies away + start music */
    wrap.classList.add('flyaway');
    if(typeof startMusic==='function')startMusic();
    /* overlay fades */
    setTimeout(()=>splash.classList.add('hide'),250);
    /* hero appears */
    setTimeout(()=>{
      document.body.style.overflow='';
      document.documentElement.style.overflow='';
      if(hero){hero.classList.remove('entering');hero.classList.add('entered')}
    },750);
    /* cleanup */
    setTimeout(()=>splash.remove(),1500);
  });
}

function plural(n,one,few,many){
  const abs=Math.abs(n)%100;const a1=abs%10;
  if(abs>10&&abs<20)return many;
  if(a1>1&&a1<5)return few;
  if(a1===1)return one;
  return many;
}

function initCountdown(){
  const target=new Date('2026-06-06T14:00:00').getTime();
  const totalMs=150*864e5; /* 150 days total */
  const sandTop=document.getElementById('sandTop');
  const sandBot=document.getElementById('sandBot');

  function updateSand(diff){
    const p=Math.min(1,Math.max(0,diff/totalMs));
    const wave=Math.sin(Date.now()/800)*2;

    /* ── TOP BULB sand ── */
    /* Sand surface y: p=1 → y=44 (full), p=0 → y=138 (empty) */
    const topY=44+(1-p)*94;
    if(p>0.02){
      /* Wide rectangle with wavy top — clipPath clips to glass shape */
      sandTop.setAttribute('d',
        'M20,'+topY+
        ' Q60,'+(topY-4+wave)+' 90,'+(topY+2+wave)+
        ' Q120,'+(topY-3-wave)+' 160,'+topY+
        ' L160,140 L20,140Z');
    } else {
      sandTop.setAttribute('d','');
    }

    /* ── BOTTOM BULB sand ── */
    /* Sand top y: p=1 → y=236 (tiny), p=0 → y=148 (full) */
    const botY=236-(1-p)*88;
    const bWave=Math.sin(Date.now()/900+1)*2;
    if(p<0.98){
      sandBot.setAttribute('d',
        'M20,'+botY+
        ' Q60,'+(botY-3+bWave)+' 90,'+(botY+1+bWave)+
        ' Q120,'+(botY-2-bWave)+' 160,'+botY+
        ' L160,240 L20,240Z');
    } else {
      sandBot.setAttribute('d','');
    }
  }

  function tick(){
    const now=Date.now(),diff=Math.max(0,target-now);
    const d=Math.floor(diff/864e5),h=Math.floor(diff%864e5/36e5),m=Math.floor(diff%36e5/6e4),s=Math.floor(diff%6e4/1e3);
    document.getElementById('cdD').textContent=String(d).padStart(2,'0');
    document.getElementById('cdH').textContent=String(h).padStart(2,'0');
    document.getElementById('cdM').textContent=String(m).padStart(2,'0');
    document.getElementById('cdS').textContent=String(s).padStart(2,'0');
    document.getElementById('cdDl').textContent=plural(d,'день','дні','днів');
    document.getElementById('cdHl').textContent=plural(h,'година','години','годин');
    document.getElementById('cdMl').textContent=plural(m,'хвилина','хвилини','хвилин');
    document.getElementById('cdSl').textContent=plural(s,'секунда','секунди','секунд');
    updateSand(diff);
  }
  tick();setInterval(tick,1000);
  /* smooth sand wave animation */
  function animateSand(){const diff=Math.max(0,target-Date.now());updateSand(diff);requestAnimationFrame(animateSand)}
  requestAnimationFrame(animateSand);
}

/* ═══ SCROLL OBSERVER ═══ */
let obs;
function initObs(){
  obs=new IntersectionObserver(en=>{en.forEach(e=>{if(e.isIntersecting)e.target.classList.add('v')})},{threshold:.1,rootMargin:'0px 0px -30px 0px'});
  document.querySelectorAll('.sf').forEach(el=>obs.observe(el));
  /* pause/resume animations per section for CPU performance */
  const secObs=new IntersectionObserver(en=>{en.forEach(e=>{
    e.target.classList.toggle('sec--visible',e.isIntersecting);
  })},{threshold:0,rootMargin:'200px'});
  document.querySelectorAll('.sec,.hero').forEach(el=>secObs.observe(el));
}

function showErr(fg,msg){clearErr(fg);const el=document.createElement('p');el.className='fg-err';el.textContent=msg;fg.appendChild(el)}
function clearErr(fg){const e=fg.querySelector('.fg-err');if(e)e.remove()}
function sendRSVP(e){
  e.preventDefault();
  const form=e.target;
  form.querySelectorAll('.fg-err').forEach(e=>e.remove());
  const ni=form.querySelector('input[type="text"]');
  const a=document.querySelector('input[name="att"]:checked');
  let ok=true;
  if(!ni.value.trim()){showErr(ni.closest('.fg'),"Будь ласка, введіть ваше ім'я");ok=false}
  if(!a){showErr(document.querySelector('.rsvp-rad').closest('.fg'),'Оберіть один із варіантів');ok=false}
  const y=a&&a.value==='yes';
  if(!ok)return;

  /* Send to Google Sheets via GET (most reliable with Apps Script) */
  const guests=document.getElementById('gCount');
  const wishes=form.querySelector('textarea');
  const guestsInfo=y&&guests?guests.value:'';
  const params=new URLSearchParams({
    name:'«'+ni.value.trim()+'»',
    attendance:y?'Так':'Ні',
    guests:guestsInfo,
    wishes:wishes&&wishes.value.trim()?'«'+wishes.value.trim()+'»':''
  });

  //   mode:'no-cors'

  document.getElementById('rsvpW').style.display='none';
